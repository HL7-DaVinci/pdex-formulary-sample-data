def build
  name.strip!

  FHIR::InsurancePlan.new(
    id: FORMULARY_ID_PREFIX + plan.id,
    meta: meta,
    text: text,
    identifier: identifier,
    status: "active",
    name: name,
    type: type,
    period: period,
  )
end
